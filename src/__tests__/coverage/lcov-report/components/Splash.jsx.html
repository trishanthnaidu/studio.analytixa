import React, { useState } from 'react';
import logo from '../assets/images/logo.svg';
import imgPrismDesc from '../assets/images/section00.svg';
import { useStyles } from '@rootzjs/ui';
import { Styles} from '../styles/Splash';
import { Checkbox } from '@rootzjs/ui';
import { Button } from '@rootzjs/ui';
import { Marker } from '@rootzjs/ui';
import { Tooltip } from '@rootzjs/ui';
import { TermsOfUse } from './TermsOfUse';
 
const Component = props => {
    const { classes, sid } = props;
    const navigateTo = sid !== "" ? "/navigate" : "/sso"
    const [termsChecked, setTermsChecked] = useState(true);
    const [showTermsOfUsePanel, setShowTermsOfUsePanel] = useState(false);
    const onChecked = disbaled => {
        setTermsChecked(!disbaled);
    }
    const onTermsOfUseClick = () => {
        setShowTermsOfUsePanel(true)
    }
    const onTermsOfUsePanelClose = () => {
        setShowTermsOfUsePanel(false)
    }
 
    return (
        <div className={styl.root}>
            {
                showTermsOfUsePanel && <TermsOfUse onCloseClick={onTermsOfUsePanelClose} isTermsChecked={termsChecked} acceptTerms={onChecked} />
            }
            <div className={styl.loginPanel}>
                <div className={styl.loginContainer}>
                    <img className={styl.logo} src={logo} alt="logo" />
                    <hr className={styl.prismUnderline} />
                    <p className={styl.desc}>Login with your organisation credentials</p>
                    <div className={styl.termsNConditionContainer}>
                        <Marker
                            show={!termsChecked}
                            title="Please accept Terms of Use to Proceed with Login"
                        >
                            <Checkbox
                                onChange={onChecked}
                                checked={termsChecked}
                                className={!termsChecked && styl.checkboxError} />
                        </Marker>
                        <span className={styl.termsNConditionText}>By logging in, I agree with &nbsp;
                            <span className={styl.termsAnchor} onClick={onTermsOfUseClick} >Terms of Use</span>
                        </span>
                    </div>
                    <div className={styl.buttonContainer}>
                        <Button
                            text="Okta"
                            className={styl.starButton}
                            disabled={!termsChecked}
                            href={navigateTo}
                        />
 
                        <Tooltip
                            title="coming soon !!"
                            className={styl.MYIDupcoming}
                            placement="top"
                        >
                            <Button
                                text="MyID"
                                color="secondary"
                                className={styl.disneyButton}
                                disabled={true}
                            />
                        </Tooltip>
                    </div>
                </div>
            </div>
            <div className={styl.newsFeedsContainer}>
                <img className={styl.imgPrismDesc} src={imgPrismDesc} alt="imgPrismDesc" />
            </div>
        </div>
    )
}
 
export const Splash = useStyles(Styles)(Component);
